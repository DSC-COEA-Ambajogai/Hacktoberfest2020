import cv2 
import numpy as np
import matplotlib.pyplot as plt

img = cv2.imread('darkImage.png', 0)

hist1 = cv2.calcHist([img], [0], None, [256], [0,256])

# Create new histogram after equalizer 

equalizedImage1 = cv2.equalizeHist(img);

equalizedHist1 = cv2.calcHist([equalizedImage1], [0], None, [256], [0, 256]) 

#plotting
plt.subplot(221), plt.plot(hist1)
plt.subplot(222), plt.plot(equalizedHist1)
plt.subplot(223), plt.imshow(cv2.cvtColor(img, cv2.COLOR_BGR2RGB))
plt.subplot(224), plt.imshow(cv2.cvtColor(equalizedImage1, cv2.COLOR_BGR2RGB))

plt.show()

# Showing image one image
result = np.hstack((img,equalizedImage1))
cv2.imshow('result', result)

cv2.waitKey(0)

import cv2 
import numpy as np
import matplotlib.pyplot as plt

image = cv2.imread("color.png", 1)

# convert image from rgb to hsv
img_hsv = cv2.cvtColor(image, cv2.COLOR_RGB2HSV)

# histogram equalization on the v-channel
img_hsv[:,:,2] = cv2.equalizeHist(img_hsv[:,:,2])

# convert the image from hsv to rgb
imgRgb = cv2.cvtColor(img_hsv, cv2.COLOR_HSV2RGB)

# histogram
hist1 = cv2.calcHist([image], [0], None, [256], [0, 256]) 
hist2 = cv2.calcHist([imgRgb], [0], None, [256], [0, 256]) 

#plotting
plt.subplot(221), plt.plot(hist1)
plt.subplot(222), plt.plot(hist2)
plt.subplot(223), plt.imshow(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
plt.subplot(224), plt.imshow(cv2.cvtColor(imgRgb, cv2.COLOR_BGR2RGB))

plt.show()
